/**
 * StaticMethodDemo - Understanding static methods
 * Baby-level: "Shared tools everyone can use without owning them"
 */
public class StaticMethodDemo {
    
    // Static methods - belong to class, not object
    static int add(int a, int b) {
        return a + b;
    }
    
    static double celsiusToFahrenheit(double celsius) {
        return (celsius * 9/5) + 32;
    }
    
    static boolean isPrime(int number) {
        if (number <= 1) return false;
        for (int i = 2; i <= Math.sqrt(number); i++) {
            if (number % i == 0) return false;
        }
        return true;
    }
    
    static void printArray(int[] arr) {
        System.out.print("[");
        for (int i = 0; i < arr.length; i++) {
            System.out.print(arr[i]);
            if (i < arr.length - 1) System.out.print(", ");
        }
        System.out.println("]");
    }
    
    public static void main(String[] args) {
        // Call static methods without creating object
        int sum = StaticMethodDemo.add(10, 20);
        System.out.println("Sum: " + sum);
        
        // Or directly (same class)
        double fahrenheit = celsiusToFahrenheit(25);
        System.out.println("25Â°C = " + fahrenheit + "Â°F");
        
        System.out.println("Is 17 prime? " + isPrime(17));
        
        int[] numbers = {1, 2, 3, 4, 5};
        printArray(numbers);
        
        // Real-world: Like Math.sqrt(), Arrays.sort() - utility functions
    }
}
Explanation: Static methods belong to the class itself, not to any object. Call them using ClassName.methodName(). Perfect for utility functions that don't need object state.

/**
 * InstanceMethodDemo - Methods that work with object data
 * Baby-level: "Each toy has its own actions"
 */
class BankAccount {
    private double balance;
    private String accountNumber;
    
    BankAccount(String accountNumber, double initialBalance) {
        this.accountNumber = accountNumber;
        this.balance = initialBalance;
    }
    
    // Instance methods - work with object's data

        if (amount > 0) {
            balance += amount;
            System.out.println("Deposited: $" + amount);

            System.out.println("Withdrawn: $" + amount);
        } else {
            System.out.println("Insufficient funds!");
        }
    }
    
    double getBalance() {
        return balance;
    }
    
    void displayAccountInfo() {
        System.out.println("Account: " + accountNumber);
        System.out.println("Balance: $" + balance);
    }
}

        // Each object has its own data and methods

        account1.deposit(500);
        account1.withdraw(200);
        account1.displayAccountInfo();
        
        System.out.println();
        
        account2.deposit(300);
        account2.displayAccountInfo();
        
        // Each account maintains its own balance independently
    }
}
Explanation: Instance methods operate on object-specific data. Each object has its own copy of instance variables, and instance methods work with that data.

Program 7: Recursive Method - Factorial

Demonstrates recursion with factorial calculation.

â˜• Java
/**
 * RecursiveFactorial - Understanding recursion
 * Baby-level: "A method that calls itself until it reaches a stopping point"
 */
public class RecursiveFactorial {
    
    // Recursive method - calls itself
    int factorial(int n) {
        // Base case - stops recursion
        if (n == 0 || n == 1) {
            return 1;
        }
        // Recursive case - calls itself with smaller value
        return n * factorial(n - 1);
    }
    
    // For comparison: iterative version
    int factorialIterative(int n) {
        int result = 1;
        for (int i = 2; i <= n; i++) {
            result *= i;
        }
        return result;
    }
    
    public static void main(String[] args) {
        RecursiveFactorial demo = new RecursiveFactorial();
        
        System.out.println("Factorial of 5 (recursive): " + demo.factorial(5));
        System.out.println("Factorial of 5 (iterative): " + demo.factorialIterative(5));
        
        // How recursion works for factorial(5):
        // factorial(5) = 5 * factorial(4)
        //              = 5 * 4 * factorial(3)
        //              = 5 * 4 * 3 * factorial(2)
        //              = 5 * 4 * 3 * 2 * factorial(1)
        //              = 5 * 4 * 3 * 2 * 1
        //              = 120
        
        System.out.println("\nFactorials 1-10:");
        for (int i = 1; i <= 10; i++) {
            System.out.println(i + "! = " + demo.factorial(i));
        }
    }
}
Explanation: Recursive methods call themselves to solve problems by breaking them into smaller sub-problems. Must have a base case to prevent infinite recursion.

Program 8: Recursive Method - Fibonacci

Demonstrates recursion with Fibonacci sequence.

â˜• Java
/**
 * RecursiveFibonacci - Fibonacci using recursion
 * Baby-level: "Each number is sum of previous two"
 */
public class RecursiveFibonacci {
    
    int fibonacci(int n) {
        // Base cases
        if (n == 0) return 0;
        if (n == 1) return 1;
        
        // Recursive case
        return fibonacci(n - 1) + fibonacci(n - 2);
    }
    
    void printFibonacciSeries(int count) {
        System.out.print("Fibonacci Series: ");
        for (int i = 0; i < count; i++) {
            System.out.print(fibonacci(i) + " ");
        }
        System.out.println();
    }
    
    public static void main(String[] args) {
        RecursiveFibonacci demo = new RecursiveFibonacci();
        
        System.out.println("10th Fibonacci number: " + demo.fibonacci(10));
        
        demo.printFibonacciSeries(15);
        
        // How it works for fibonacci(5):
        // fib(5) = fib(4) + fib(3)
        //        = (fib(3) + fib(2)) + (fib(2) + fib(1))
        //        = ((fib(2) + fib(1)) + (fib(1) + fib(0))) + ...
        // Eventually reaches base cases and calculates back up
    }
}
Explanation: Fibonacci demonstrates recursion with multiple recursive calls. Each call branches into two more calls until base cases are reached.

/**

 * Baby-level: "Different sized spoons, all called 'spoon'"
 */
public class MethodOverloadingDemo {
    
    // Overloaded add methods - different parameter count
    int add(int a, int b) {
        System.out.println("Adding two integers");
        return a + b;
    }
    
    int add(int a, int b, int c) {
        System.out.println("Adding three integers");
        return a + b + c;
    }
    
    // Overloaded add methods - different parameter types
    double add(double a, double b) {
        System.out.println("Adding two doubles");
        return a + b;
    }
    
    String add(String a, String b) {
        System.out.println("Concatenating two strings");
        return a + b;
    }
    
    // Overloaded display methods
    void display(int num) {
        System.out.println("Integer: " + num);
    }
    
    void display(String text) {
        System.out.println("String: " + text);
    }
    
    void display(int num, String text) {
        System.out.println("Number: " + num + ", Text: " + text);
    }

        System.out.println("Result: " + demo.add(5, 10));

Explanation: Method overloading allows multiple methods with the same name but different parameters. Java chooses the correct method based on the arguments at compile-time.

/**

class Student {
    private String name;
    private int age;
    private double gpa;
    
    // Getter for name
    public String getName() {
        return name;
    }
    
    // Setter for name with validation

        if (name != null && !name.trim().isEmpty()) {

        }
    }
    
    // Getter for age
    public int getAge() {
        return age;
    }

        if (age > 0 && age < 150) {
            this.age = age;
        } else {
            System.out.println("Invalid age!");
        }
    }
    
    // Getter for GPA
    public double getGpa() {
        return gpa;
    }

            this.gpa = gpa;
        } else {
            System.out.println("Invalid GPA! Must be 0.0-4.0");
        }
    }
    
    public void displayInfo() {
        System.out.println("Name: " + name);
        System.out.println("Age: " + age);
        System.out.println("GPA: " + gpa);
    }

Program 14: Method Returning Object

Demonstrates methods that create and return objects.

â˜• Java
/**
 * MethodReturningObject - Methods can return objects
 * Baby-level: "Method creates and gives back a complete package"
 */
class Rectangle {
    int width;
    int height;
    
    Rectangle(int width, int height) {
        this.width = width;
        this.height = height;
    }
    
    int getArea() {
        return width * height;
    }
    
    void display() {
        System.out.println("Rectangle: " + width + "x" + height + 
                         " (Area: " + getArea() + ")");
    }
}

class RectangleFactory {
    
    // Method returns a new Rectangle object
    Rectangle createSquare(int side) {
        return new Rectangle(side, side);
    }
    
    Rectangle createRectangle(int width, int height) {
        return new Rectangle(width, height);
    }
    
    // Method returns larger of two rectangles
    Rectangle getLarger(Rectangle r1, Rectangle r2) {
        if (r1.getArea() > r2.getArea()) {
            return r1;
        } else {
            return r2;
        }
    }
}

public class MethodReturningObject {
    public static void main(String[] args) {
        RectangleFactory factory = new RectangleFactory();
        
        // Methods return Rectangle objects
        Rectangle square = factory.createSquare(5);
        square.display();
        
        Rectangle rect = factory.createRectangle(10, 3);
        rect.display();
        
        Rectangle larger = factory.getLarger(square, rect);
        System.out.print("Larger: ");
        larger.display();
    }

Complete real-world example: Loan EMI calculation system.

/**
 * EMICalculator - Real-world loan EMI calculator
 * Baby-level: "Calculate monthly payment for a loan"
 */
class EMICalculator {
    
    // Calculate monthly EMI
    double calculateEMI(double principal, double annualRate, int years) {
        double monthlyRate = annualRate / 12 / 100;
        int months = years * 12;
        
        double emi = (principal * monthlyRate * Math.pow(1 + monthlyRate, months)) /
                     (Math.pow(1 + monthlyRate, months) - 1);
        
        return Math.round(emi * 100.0) / 100.0;  // Round to 2 decimals
    }
    
    // Calculate total amount payable
    double calculateTotalAmount(double emi, int years) {
        return emi * years * 12;
    }
    
    // Calculate total interest
    double calculateTotalInterest(double totalAmount, double principal) {
        return totalAmount - principal;
    }
    
    // Validate loan parameters
    boolean validateLoan(double principal, double rate, int years) {
        return principal > 0 && rate >= 0 && years > 0;
    }
    
    // Display complete loan details
    void displayLoanDetails(double principal, double annualRate, int years) {
        if (!validateLoan(principal, annualRate, years)) {
            System.out.println("Invalid loan parameters!");
            return;
        }
        
        System.out.println("=== Loan EMI Calculator ===");
        System.out.println("Principal Amount: $" + principal);
        System.out.println("Annual Interest Rate: " + annualRate + "%");
        System.out.println("Loan Tenure: " + years + " years");
        System.out.println();
        
        double emi = calculateEMI(principal, annualRate, years);
        System.out.println("Monthly EMI: $" + emi);
        
        double totalAmount = calculateTotalAmount(emi, years);
        System.out.println("Total Amount Payable: $" + totalAmount);
        
        double totalInterest = calculateTotalInterest(totalAmount, principal);
        System.out.println("Total Interest: $" + totalInterest);
        System.out.println("==========================");
    }
    
    public static void main(String[] args) {
        EMICalculator calculator = new EMICalculator();
        
        // Home loan example
        calculator.displayLoanDetails(500000, 7.5, 20);
        
        System.out.println();
        
        // Car loan example
        calculator.displayLoanDetails(30000, 9.0, 5);
    }
}
