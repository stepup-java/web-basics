Q1. Build a Simple Calculator (Console Based).

Create a program that takes two numbers and an operator (+, -, *, /) from user and performs the calculation.


                    // Sample Input:
                    // 10 20
                    // +
                    // Sample Output:
                    // Result: 30.0
üí° Show Answer 

import java.util.Scanner;
public class Calculator {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter two numbers: ");
        double a = sc.nextDouble();
        double b = sc.nextDouble();
        System.out.print("Enter operator (+, -, *, /): ");
        char op = sc.next().charAt(0);
        double result = 0;
        switch(op) {
            case '+': result = a + b; break;
            case '-': result = a - b; break;
            case '*': result = a * b; break;
            case '/': result = a / b; break;
            default: System.out.println("Invalid operator"); return;
        }
        System.out.println("Result: " + result);
    }
}

Q2. Create a Student Grade Tracker.

Input marks for 5 subjects, calculate total, percentage, and assign a grade (A, B, C, D, F).


                    // Sample Input:
                    // 85 90 78 92 88
                    // Sample Output:
                    // Total: 433, Percentage: 86.6%, Grade: B
üí° Show Answer 

import java.util.Scanner;
public class GradeTracker {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int total = 0;
        for(int i=1; i<=5; i++) {
            System.out.print("Enter marks for subject " + i + ": ");
            total += sc.nextInt();
        }
        double percentage = total / 5.0;
        char grade = (percentage >= 90) ? 'A' : (percentage >= 80) ? 'B' : (percentage >= 70) ? 'C' : (percentage >= 60) ? 'D' : 'F';
        System.out.println("Total: " + total + ", Percentage: " + percentage + "%, Grade: " + grade);
    }
}

Q3. Implement a Bank Account System.

Create a class `BankAccount` with methods to deposit, withdraw, and check balance. Ensure balance doesn't go negative.


                    // Sample Output:
                    // Deposited: 1000.0
                    // Withdrawn: 500.0
                    // Balance: 500.0
üí° Show Answer 

class BankAccount {
    private double balance;
    public void deposit(double amount) { balance += amount; System.out.println("Deposited: " + amount); }
    public void withdraw(double amount) {
        if(balance >= amount) { balance -= amount; System.out.println("Withdrawn: " + amount); }
        else System.out.println("Insufficient funds");
    }
    public double getBalance() { return balance; }
}
public class Main {
    public static void main(String[] args) {
        BankAccount acc = new BankAccount();
        acc.deposit(1000);
        acc.withdraw(500);
        System.out.println("Balance: " + acc.getBalance());
    }
}

Q4. Develop a Number Guessing Game.

Computer picks a random number between 1-100. User tries to guess it. Hints: "Too High" or "Too Low".


                    // Sample Input:
                    // 50
                    // Sample Output:
                    // Too High
üí° Show Answer 

import java.util.Scanner;
import java.util.Random;
public class GuessGame {
    public static void main(String[] args) {
        int number = new Random().nextInt(100) + 1;
        Scanner sc = new Scanner(System.in);
        int guess;
        do {
            System.out.print("Guess the number (1-100): ");
            guess = sc.nextInt();
            if (guess < number) System.out.println("Too Low");
            else if (guess > number) System.out.println("Too High");
        } while (guess != number);
        System.out.println("Correct! You guessed it.");
    }
}

Q5. Create an Employee Payroll System.

Calculate Gross Salary = Basic + HRA (20%) + DA (50%). Deduct PF (12%). Print Net Salary.


                    // Sample Input:
                    // 50000
                    // Sample Output:
                    // Gross Salary: 85000.0
                    // Net Salary: 79000.0
üí° Show Answer 

import java.util.Scanner;
public class Payroll {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter Basic Salary: ");
        double basic = sc.nextDouble();
        double hra = 0.20 * basic;
        double da = 0.50 * basic;
        double pf = 0.12 * basic;
        double gross = basic + hra + da;
        double net = gross - pf;
        System.out.println("Gross Salary: " + gross);
        System.out.println("Net Salary: " + net);
    }
}

Q8. Simulate a Dice Rolling Game.

Roll two dice 10 times. Count how many times doubles (same number on both) appear.


                    // Sample Output:
                    // Rolled: 3, 4
                    // ...
                    // Doubles count: 2
üí° Show Answer 

import java.util.Random;
public class DiceGame {
    public static void main(String[] args) {
        Random rand = new Random();
        int doubles = 0;
        for(int i=0; i<10; i++) {
            int d1 = rand.nextInt(6) + 1;
            int d2 = rand.nextInt(6) + 1;
            System.out.println("Rolled: " + d1 + ", " + d2);
            if(d1 == d2) doubles++;
        }
        System.out.println("Doubles count: " + doubles);
    }
}

Q9. Check for Palindrome Sentence.

                    // Sample Output:
                    // Is Palindrome: true

public class Palindrome {
    public static void main(String[] args) {
        String s = "Was it a car or a cat I saw";
        String clean = s.replaceAll("\\s+", "").toLowerCase();
        String rev = new StringBuilder(clean).reverse().toString();
        System.out.println("Is Palindrome: " + clean.equals(rev));
    }
}

public class PasswordGen {

        String chars = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789!@#$%";

Q16. Count Word Frequency in a Text File.

                    // Sample Output:

import java.util.*;
import java.nio.file.*;
public class WordCount {
    public static void main(String[] args) throws Exception {
        String content = new String(Files.readAllBytes(Paths.get("test.txt")));
        Map map = new HashMap<>();
        for(String w : content.split("\\s+")) {
            map.put(w, map.getOrDefault(w, 0) + 1);
